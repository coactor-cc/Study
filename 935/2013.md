二叉查找过程中，若S[i+1]<S[i]，意味着key<S[i];反之，意味着key>S[i];
```c++
int IsKey(int S[],int n){//暴力解
    for(int i=0;i<n-1;i++){
        if(S[i+1]<S[i]){
            for(int j=i+2;j<n;j++)
                if(S[j]>S[i])
                    return 0;
        }
        else{
            for(int j=i+2;j<n;j++)
                if(S[j]<S[i])
                    return 0;
        }
    }
    return 1;
}
```
